#Install the docker pipeline plugin
#Add the jenkins user to the docker group
#Restart jenkins after adding the user

#Create a pipeline with the code:
pipeline {
    agent { docker { image 'node:16.13.1-alpine' } }
    stages {
        stage('build') {
            steps {
                sh 'node --version'
            }
        }
    }
}

#In the general jenkins settings you can set Docker Label so that pipelines with agent docker are run on hosts with this label(s)
