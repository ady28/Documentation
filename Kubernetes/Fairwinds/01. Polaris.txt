#To run Polaris as a dashboard (no outside traffice enabled)
kubectl apply -f polaris-dashboard-config.yaml
kubectl apply -f polaris-dashboard.yaml
#For external access either use a NodePort, use proxy or go through an ingress
#Create a user and password for authentication to dashboard
htpasswd -nb root root | openssl base64
#Place the output in secret.yaml
kubectl apply -f secret.yaml
kubectl apply -f mw.yaml
kubectl apply -f ingress.yaml

#If the config map is changed it will be changed automatically in the volume but a restart of the application is needed
kubectl rollout restart deployment.apps/polaris-dashboard -n polaris
