#List block devices
lsblk
fdisk -l

#Initialize disk and create partition
fdisk /dev/sdb
#Press n, then p, then 1, leave default for sector data, then press w
#Create a filesystem on device
mkfs.ext4 /dev/sdb1
#Create a mountpoint
cd /mnt
mkdir backups
#Mount device
mount /dev/sdb1 /mnt/backups
#Make this persistent
nano /etc/fstab
#Add a line:
/dev/sdb1	/mnt/backups	ext4	defaults	0	0

#List mounted devices
df -h
#Unmount a device
umount /dev/sdb1
